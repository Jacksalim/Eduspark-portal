# EduSpark Quiz System — Frontend Integration Guide

## Overview

This guide shows how to integrate the AI quiz generation API with your EduSpark frontend React components.

## Quick Integration

### 1. Create a Quiz Hook

Create `src/hooks/useQuizGenerator.jsx`:

```jsx
import { useState, useCallback } from 'react'

export function useQuizGenerator() {
  const [loading, setLoading] = useState(false)
  const [error, setError] = useState(null)
  const [questions, setQuestions] = useState([])
  const [provider, setProvider] = useState(null)
  const [responseTime, setResponseTime] = useState(null)

  const generateQuiz = useCallback(async (grade, subject) => {
    setLoading(true)
    setError(null)
    setQuestions([])
    const startTime = Date.now()

    try {
      const response = await fetch('/api/quiz', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
        },
        body: JSON.stringify({ grade, subject }),
      })

      const duration = Date.now() - startTime
      setResponseTime(duration)

      const data = await response.json()

      if (!response.ok) {
        // Handle error
        const userMessage = data.userMessage || data.error || 'Failed to generate quiz'
        setError(userMessage)
        console.error('Quiz generation error:', data)
        return { success: false, error: userMessage }
      }

      // Success
      setQuestions(data.questions || [])
      setProvider(data.provider || 'unknown')
      
      console.log(`Quiz generated by ${data.provider} in ${duration}ms`)
      return { success: true, questions: data.questions, provider: data.provider }

    } catch (err) {
      console.error('Network error:', err)
      const errorMessage = err.message || 'Network error. Please try again.'
      setError(errorMessage)
      return { success: false, error: errorMessage }
    } finally {
      setLoading(false)
    }
  }, [])

  const reset = useCallback(() => {
    setLoading(false)
    setError(null)
    setQuestions([])
    setProvider(null)
    setResponseTime(null)
  }, [])

  return {
    generateQuiz,
    loading,
    error,
    questions,
    provider,
    responseTime,
    reset,
  }
}
```

### 2. Create a Quiz Generator Component

Create `src/components/QuizGenerator.jsx`:

```jsx
import { useState } from 'react'
import { useQuizGenerator } from '../hooks/useQuizGenerator'

const GRADES = Array.from({ length: 12 }, (_, i) => i + 1)
const SUBJECTS = [
  'Mathematics',
  'English',
  'Science',
  'Social Studies',
  'ICT',
]

export function QuizGenerator() {
  const [selectedGrade, setSelectedGrade] = useState(9)
  const [selectedSubject, setSelectedSubject] = useState('Mathematics')
  
  const {
    generateQuiz,
    loading,
    error,
    questions,
    provider,
    responseTime,
    reset,
  } = useQuizGenerator()

  const handleGenerate = async () => {
    await generateQuiz(selectedGrade, selectedSubject)
  }

  return (
    <div className="quiz-generator">
      <div className="quiz-controls">
        <div className="control-group">
          <label htmlFor="grade">Grade:</label>
          <select
            id="grade"
            value={selectedGrade}
            onChange={(e) => setSelectedGrade(Number(e.target.value))}
            disabled={loading}
          >
            {GRADES.map((g) => (
              <option key={g} value={g}>
                Grade {g}
              </option>
            ))}
          </select>
        </div>

        <div className="control-group">
          <label htmlFor="subject">Subject:</label>
          <select
            id="subject"
            value={selectedSubject}
            onChange={(e) => setSelectedSubject(e.target.value)}
            disabled={loading}
          >
            {SUBJECTS.map((s) => (
              <option key={s} value={s}>
                {s}
              </option>
            ))}
          </select>
        </div>

        <button
          onClick={handleGenerate}
          disabled={loading}
          className={`btn-generate ${loading ? 'loading' : ''}`}
        >
          {loading ? 'Generating Quiz...' : 'Generate Quiz'}
        </button>

        {questions.length > 0 && (
          <button onClick={reset} className="btn-reset">
            Clear
          </button>
        )}
      </div>

      {error && (
        <div className="error-message">
          <strong>Error:</strong> {error}
          <p className="error-hint">Try again in a moment or contact support.</p>
        </div>
      )}

      {responseTime && (
        <div className="metadata">
          <small>
            Generated by <strong>{provider}</strong> in <strong>{responseTime}ms</strong>
          </small>
        </div>
      )}

      {questions.length > 0 && (
        <div className="quiz-questions">
          {questions.map((q, index) => (
            <div key={index} className="question-card">
              <div className="question-number">Question {index + 1} of {questions.length}</div>
              <h3 className="question-text">{q.question}</h3>
              
              <div className="options">
                {q.options.map((option, optIndex) => (
                  <label key={optIndex} className="option">
                    <input
                      type="radio"
                      name={`question-${index}`}
                      value={option}
                    />
                    <span>{option}</span>
                  </label>
                ))}
              </div>

              <details className="explanation">
                <summary>Show Explanation</summary>
                <p>{q.explanation}</p>
              </details>
            </div>
          ))}

          <div className="quiz-actions">
            <button className="btn-submit">Submit Quiz</button>
            <button className="btn-download">Download PDF</button>
          </div>
        </div>
      )}
    </div>
  )
}
```

### Add Quiz Submission

```jsx
async function submitQuiz(answers) {
  // Calculate score
  let score = 0
  answers.forEach((answer, index) => {
    if (answer === questions[index].correctAnswer) {
      score++
    }
  })

  const percentage = (score / questions.length) * 100

  console.log(`Score: ${score}/${questions.length} (${percentage.toFixed(1)}%)`)

  // Save to Supabase
  if (supabaseClient) {
    await supabaseClient
      .from('quiz_results')
      .insert({
        user_id: userId,
        grade: selectedGrade,
        subject: selectedSubject,
        score,
        total: questions.length,
        answers,
        created_at: new Date().toISOString(),
      })
  }
}
```

### Add PDF Export

```jsx
import jsPDF from 'jspdf'

function exportToPDF() {
  const pdf = new jsPDF()
  let yPosition = 20

  pdf.setFontSize(16)
  pdf.text(`${selectedSubject} Quiz - Grade ${selectedGrade}`, 20, yPosition)
  yPosition += 15

  pdf.setFontSize(10)
  questions.forEach((q, index) => {
    if (yPosition > 250) {
      pdf.addPage()
      yPosition = 20
    }

    pdf.text(`Q${index + 1}: ${q.question}`, 20, yPosition)
    yPosition += 10

    q.options.forEach((opt) => {
      pdf.text(`  ☐ ${opt}`, 25, yPosition)
      yPosition += 7
    })

    yPosition += 5
  })

  pdf.save(`quiz-${selectedSubject}-grade-${selectedGrade}.pdf`)
}
```

## Error Handling Best Practices

```jsx
function getErrorMessage(error) {
  if (error.includes('timeout')) {
    return 'Quiz generation is taking longer than expected. Please try again.'
  }
  if (error.includes('No AI providers')) {
    return 'Service is currently unavailable. Please try again later.'
  }
  if (error.includes('Invalid input')) {
    return 'Please select a valid grade and subject.'
  }
  if (error.includes('Network')) {
    return 'Network error. Please check your connection and try again.'
  }
  return error
}
```
